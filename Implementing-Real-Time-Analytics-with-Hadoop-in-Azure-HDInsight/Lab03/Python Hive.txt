# Create a dataframe from a Hive query
calls = spark.sql("""SELECT devicemodel, COUNT(*) AS calls
   FROM hivesampletable
   GROUP BY devicemodel
   ORDER BY calls DESC""")
calls.show()


# Display schema
calls.printSchema()
